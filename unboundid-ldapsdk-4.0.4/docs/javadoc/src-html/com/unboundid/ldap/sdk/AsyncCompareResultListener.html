021package com.unboundid.ldap.sdk;
022
023
024
025import com.unboundid.util.Extensible;
026import com.unboundid.util.ThreadSafety;
027import com.unboundid.util.ThreadSafetyLevel;
028
029
030
031/**
032 * This interface provides a mechanism for notifying a client when the result
033 * for an asynchronous compare operation has been received.
034 */
035@Extensible()
036@ThreadSafety(level=ThreadSafetyLevel.INTERFACE_NOT_THREADSAFE)
037public interface AsyncCompareResultListener
038{
039  /**
040   * Indicates that the provided LDAP result has been received in response to an
041   * asynchronous compare operation.  Note that automatic referral following is
042   * not supported for asynchronous operations, so it is possible that this
043   * result could include a referral.
044   *
045   * @param  requestID      The async request ID of the request for which the
046   *                        response was received.
047   * @param  compareResult  The compare result that has been received.
048   */
049  void compareResultReceived(AsyncRequestID requestID,
050                             CompareResult compareResult);
051}
