060/**
061 * This class provides a simple tool that can be used to search an LDAP
062 * directory server.  Some of the APIs demonstrated by this example include:
063 * <UL>
064 *   <LI>Argument Parsing (from the {@code com.unboundid.util.args}
065 *       package)</LI>
066 *   <LI>LDAP Command-Line Tool (from the {@code com.unboundid.util}
067 *       package)</LI>
068 *   <LI>LDAP Communication (from the {@code com.unboundid.ldap.sdk}
069 *       package)</LI>
070 * </UL>
071 * <BR><BR>
072 * All of the necessary information is provided using
073 * command line arguments.  Supported arguments include those allowed by the
074 * {@link LDAPCommandLineTool} class, as well as the following additional
075 * arguments:
076 * <UL>
077 *   <LI>"-b {baseDN}" or "--baseDN {baseDN}" -- specifies the base DN to use
078 *       for the search.  This must be provided.</LI>
079 *   <LI>"-s {scope}" or "--scope {scope}" -- specifies the scope to use for the
080 *       search.  The scope value should be one of "base", "one", "sub", or
081 *       "subord".  If this isn't specified, then a scope of "sub" will be
082 *       used.</LI>
083 *   <LI>"-R" or "--followReferrals" -- indicates that the tool should follow
084 *       any referrals encountered while searching.</LI>
085 *   <LI>"-t" or "--terse" -- indicates that the tool should generate minimal
086 *       output beyond the search results.</LI>
087 *   <LI>"-i {millis}" or "--repeatIntervalMillis {millis}" -- indicates that
088 *       the search should be periodically repeated with the specified delay
089 *       (in milliseconds) between requests.</LI>
090 *   <LI>"-n {count}" or "--numSearches {count}" -- specifies the total number
091 *       of times that the search should be performed.  This may only be used in
092 *       conjunction with the "--repeatIntervalMillis" argument.  If
093 *       "--repeatIntervalMillis" is used without "--numSearches", then the
094 *       searches will continue to be repeated until the tool is
095 *       interrupted.</LI>
096 *   <LI>"--bindControl {control}" -- specifies a control that should be
097 *       included in the bind request sent by this tool before performing any
098 *       search operations.</LI>
099 *   <LI>"-J {control}" or "--control {control}" -- specifies a control that
100 *       should be included in the search request(s) sent by this tool.</LI>
101 * </UL>
102 * In addition, after the above named arguments are provided, a set of one or
103 * more unnamed trailing arguments must be given.  The first argument should be
104 * the string representation of the filter to use for the search.  If there are
105 * any additional trailing arguments, then they will be interpreted as the
106 * attributes to return in matching entries.  If no attribute names are given,
107 * then the server should return all user attributes in matching entries.
108 * <BR><BR>
109 * Note that this class implements the SearchResultListener interface, which
110 * will be notified whenever a search result entry or reference is returned from
111 * the server.  Whenever an entry is received, it will simply be printed
112 * displayed in LDIF.
113 *
114 * @see  com.unboundid.ldap.sdk.unboundidds.tools.LDAPSearch
115 */
116@ThreadSafety(level=ThreadSafetyLevel.NOT_THREADSAFE)
117public final class LDAPSearch
118       extends LDAPCommandLineTool
119       implements SearchResultListener

420   * generic {@code LDAPCommandLineTool} framework.
421   *

430    this.parser = parser;
431
432    String description = "The base DN to use for the search.  This must be " +

435    baseDN.addLongIdentifier("base-dn", true);
436    parser.addArgument(baseDN);
437
438
439    description = "The scope to use for the search.  It should be 'base', " +
440                  "'one', 'sub', or 'subord'.  If this is not provided, then " +
441                  "a default scope of 'sub' will be used.";
442    scopeArg = new ScopeArgument('s', "scope", false, "{scope}", description,
443                                 SearchScope.SUB);
444    parser.addArgument(scopeArg);
445
446
447    description = "Follow any referrals encountered during processing.";
448    followReferrals = new BooleanArgument('R', "followReferrals", description);
449    followReferrals.addLongIdentifier("follow-referrals", true);
450    parser.addArgument(followReferrals);
451
452
453    description = "Information about a control to include in the bind request.";
454    bindControls = new ControlArgument(null, "bindControl", false, 0, null,
455         description);
456    bindControls.addLongIdentifier("bind-control", true);
457    parser.addArgument(bindControls);
458
459
460    description = "Information about a control to include in search requests.";
461    searchControls = new ControlArgument('J', "control", false, 0, null,
462         description);
463    parser.addArgument(searchControls);
464
465
466    description = "Generate terse output with minimal additional information.";
467    terseMode = new BooleanArgument('t', "terse", description);
468    parser.addArgument(terseMode);
469
470
471    description = "Specifies the length of time in milliseconds to sleep " +
472                  "before repeating the same search.  If this is not " +
473                  "provided, then the search will only be performed once.";
474    repeatIntervalMillis = new IntegerArgument('i', "repeatIntervalMillis",
475                                               false, 1, "{millis}",
476                                               description, 0,
477                                               Integer.MAX_VALUE);
478    repeatIntervalMillis.addLongIdentifier("repeat-interval-millis", true);
479    parser.addArgument(repeatIntervalMillis);
480
481
482    description = "Specifies the number of times that the search should be " +
483                  "performed.  If this argument is present, then the " +
484                  "--repeatIntervalMillis argument must also be provided to " +
485                  "specify the length of time between searches.  If " +
486                  "--repeatIntervalMillis is used without --numSearches, " +
487                  "then the search will be repeated until the tool is " +
488                  "interrupted.";
489    numSearches = new IntegerArgument('n', "numSearches", false, 1, "{count}",
490                                      description, 1, Integer.MAX_VALUE);
491    numSearches.addLongIdentifier("num-searches", true);
492    parser.addArgument(numSearches);
493    parser.addDependentArgumentSet(numSearches, repeatIntervalMillis);
494  }

498  /**
499   * {@inheritDoc}
500   */
501  @Override()
502  public void doExtendedNonLDAPArgumentValidation()
503         throws ArgumentException
504  {
505    // There must have been at least one trailing argument provided, and it must
506    // be parsable as a valid search filter.
507    if (parser.getTrailingArguments().isEmpty())
508    {
509      throw new ArgumentException("At least one trailing argument must be " +
510           "provided to specify the search filter.  Additional trailing " +
511           "arguments are allowed to specify the attributes to return in " +
512           "search result entries.");
513    }
514
515    try
516    {
517      Filter.create(parser.getTrailingArguments().get(0));
518    }
519    catch (final Exception e)
520    {
521      Debug.debugException(e);
522      throw new ArgumentException(
523           "The first trailing argument value could not be parsed as a valid " +
524                "LDAP search filter.",
525           e);
526    }
527  }

542  /**
543   * Performs the actual processing for this tool.  In this case, it gets a
544   * connection to the directory server and uses it to perform the requested
545   * search.
546   *
547   * @return  The result code for the processing that was performed.
548   */
549  @Override()
550  public ResultCode doToolProcessing()
551  {
552    // Make sure that at least one trailing argument was provided, which will be
553    // the filter.  If there were any other arguments, then they will be the
554    // attributes to return.
555    final List<String> trailingArguments = parser.getTrailingArguments();
556    if (trailingArguments.isEmpty())
557    {
558      err("No search filter was provided.");
559      err();
560      err(parser.getUsageString(StaticUtils.TERMINAL_WIDTH_COLUMNS - 1));
561      return ResultCode.PARAM_ERROR;
562    }
563
564    final Filter filter;
565    try
566    {
567      filter = Filter.create(trailingArguments.get(0));
568    }
569    catch (final LDAPException le)
570    {
571      err("Invalid search filter:  ", le.getMessage());
572      return le.getResultCode();
573    }
574
575    final String[] attributesToReturn;
576    if (trailingArguments.size() > 1)
577    {
578      attributesToReturn = new String[trailingArguments.size() - 1];
579      for (int i=1; i < trailingArguments.size(); i++)
580      {
581        attributesToReturn[i-1] = trailingArguments.get(i);
582      }
583    }
584    else
585    {
586      attributesToReturn = StaticUtils.NO_STRINGS;
587    }
588
589
590    // Get the connection to the directory server.
591    final LDAPConnection connection;
592    try
593    {
594      connection = getConnection();
595      if (! terseMode.isPresent())
596      {
597        out("# Connected to ", connection.getConnectedAddress(), ':',
598             connection.getConnectedPort());
599      }
600    }
601    catch (final LDAPException le)
602    {
603      err("Error connecting to the directory server:  ", le.getMessage());
604      return le.getResultCode();
605    }
606
607
608    // Create a search request with the appropriate information and process it
609    // in the server.  Note that in this case, we're creating a search result
610    // listener to handle the results since there could potentially be a lot of
611    // them.
612    final SearchRequest searchRequest =
613         new SearchRequest(this, baseDN.getStringValue(), scopeArg.getValue(),
614                           DereferencePolicy.NEVER, 0, 0, false, filter,
615                           attributesToReturn);
616    searchRequest.setFollowReferrals(followReferrals.isPresent());
617
618    final List<Control> controlList = searchControls.getValues();
619    if (controlList != null)
620    {
621      searchRequest.setControls(controlList);
622    }
623
624
625    final boolean infinite;
626    final int numIterations;
627    if (repeatIntervalMillis.isPresent())
628    {
629      repeat = true;
630
631      if (numSearches.isPresent())
632      {
633        infinite      = false;
634        numIterations = numSearches.getValue();
635      }
636      else
637      {
638        infinite      = true;
639        numIterations = Integer.MAX_VALUE;
640      }
641    }
642    else
643    {
644      infinite      = false;
645      repeat        = false;
646      numIterations = 1;
647    }
648
649    ResultCode resultCode = ResultCode.SUCCESS;
650    long lastSearchTime = System.currentTimeMillis();
651    final WakeableSleeper sleeper = new WakeableSleeper();
652    for (int i=0; (infinite || (i < numIterations)); i++)
653    {
654      if (repeat && (i > 0))
655      {
656        final long sleepTime =
657             (lastSearchTime + repeatIntervalMillis.getValue()) -
658             System.currentTimeMillis();
659        if (sleepTime > 0)
660        {
661          sleeper.sleep(sleepTime);
662        }
663        lastSearchTime = System.currentTimeMillis();
664      }
665
666      try
667      {
668        final SearchResult searchResult = connection.search(searchRequest);
669        if ((! repeat) && (! terseMode.isPresent()))
670        {
671          out("# The search operation was processed successfully.");
672          out("# Entries returned:  ", searchResult.getEntryCount());
673          out("# References returned:  ", searchResult.getReferenceCount());
674        }
675      }
676      catch (final LDAPException le)
677      {
678        err("An error occurred while processing the search:  ",
679             le.getMessage());
680        err("Result Code:  ", le.getResultCode().intValue(), " (",
681             le.getResultCode().getName(), ')');
682        if (le.getMatchedDN() != null)
683        {
684          err("Matched DN:  ", le.getMatchedDN());
685        }
686
687        if (le.getReferralURLs() != null)
688        {
689          for (final String url : le.getReferralURLs())
690          {
691            err("Referral URL:  ", url);
692          }
693        }
694
695        if (resultCode == ResultCode.SUCCESS)
696        {
697          resultCode = le.getResultCode();
698        }
699
700        if (! le.getResultCode().isConnectionUsable())
701        {
702          break;
703        }
704      }
705    }
706
707
708    // Close the connection to the directory server and exit.
709    connection.close();
710    if (! terseMode.isPresent())
711    {
712      out();
713      out("# Disconnected from the server");
714    }
715    return resultCode;
716  }
